# 复制以下内容到你的 .env.local 文件中

# ========== AI 模型配置 ==========

# 选择使用的AI服务：'openai' 或 'ollama' 或 'claude'
AI_PROVIDER=ollama

# OpenAI API（如果使用OpenAI）
OPENAI_API_KEY=sk-proj-xxxxx

# Ollama 配置（如果使用Ollama）
OLLAMA_BASE_URL=http://localhost:11434
OLLAMA_MODEL=gpt-oss:20b

# Claude API（如果使用Claude）
ANTHROPIC_API_KEY=sk-ant-api03-...

# ========== 其他API配置 ==========

# Google Custom Search API
GOOGLE_API_KEY=在这里粘贴你的Google_API密钥
GOOGLE_CSE_ID=在这里粘贴你的搜索引擎ID

# Judge0 Code Execution API  
JUDGE0_API_KEY=在这里粘贴你的RapidAPI密钥
